# ライブラリをインストール

yarn add @iktakahiro/markdown-it-katex --dev

# .vuepress/config.jsの書き換え

以下のコードを動画を貼りたい場所に記述します．

head: [
      ['link', {
          rel: 'stylesheet',
          href: 'https://cdn.jsdelivr.net/npm/katex@0.11.0/dist/katex.min.css'
      }]
  ],
  markdown: {
      extendMarkdown: md => {
          md.use(require('@iktakahiro/markdown-it-katex'), {
              throwOnError: false,
              errorColor: "#cc0000",
              macros: {
                  '\\Z': '\\mathbb{Z}',
                  '*': '\\times'
              }
          })
      }
  },
  (省略)
